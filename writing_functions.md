``` r
z_scores = function(x) {
        
  if (!is.numeric(x)) {
     stop("Argument x should be numeric")      
  }        
  
  if (length(x) < 3) {
     stop("Input must have at least 3 numbers")
  }     
  z = (x - mean(x)) / sd(x)
  
  return(z)
              
}
```
